====test====
A defined 401 response is returned when security validation is not passed.
====spec====
openapi: 3.0.1
paths:
  /users:
    get:
      description: returns users
      responses:
        200:
          description: ok
        401:
          $ref: '#/components/responses/InvalidAuthorization'
components:
  responses:
    InvalidAuthorization:
      description: InvalidAuthorization
      content:
        application/json:
          example: abc
  securitySchemes:
    Basic:
      type: http
      scheme: basic
security:
  - Basic: []
====server====
mock -p 4010
====command====
curl -i http://localhost:4010/users
====expect====
HTTP/1.1 401 Unauthorized
content-type: application/json; charset=utf-8

"abc"
