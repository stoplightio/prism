====test====
Adding a required header of "Authorization: Bearer <<BEARER_TOKEN>>" results in granted access.

paths:
  /todos:
    get:
      responses:
        200:
          description: OK
          content:
            application/json:
              example: test

  securitySchemes:
    BearerAuth:
      type: http
      scheme: bearer

  - BearerAuth: []

mock -p 4010 ${document}
