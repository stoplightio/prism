====test====
Providing an api key that is required to be in the query, results in granted access.
====spec====
swagger: '2.0'
paths:
  /todos:
    get:
      produces:
        - application/json
      responses:
        200:
          description: OK
          examples:
            application/json: test
securityDefinitions:
  ApiKeyAuth:
    type: apiKey
    in: query
    name: apiKey
security:
  - ApiKeyAuth: []
====server====
mock -p 4010 ${document}
====command====
curl -i http://localhost:4010/todos?apiKey=foo
====expect====
HTTP/1.1 200 OK

"test"
