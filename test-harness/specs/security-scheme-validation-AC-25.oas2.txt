====test====
Providing just one of the two required API KEYS denies access to a resource.
====spec====
swagger: '2.0'
paths:
  /todos:
    get:
      produces:
        - application/json
      responses:
        200:
          description: OK
          examples:
            application/json: test
securityDefinitions:
  ApiKeyOne:
    type: apiKey
    in: header
    name: X-API-Key
  ApiKeyTwo:
    type: apiKey
    in: query
    name: apiKey
security:
  - ApiKeyOne: []
    ApiKeyTwo: []
====server====
mock -p 4010
====command====
curl -i http://localhost:4010/todos?apiKey=abc123
====expect====
HTTP/1.1 401 Unauthorized

{"type":"Unauthorised","title":"Invalid security scheme used","status":401,"detail":""}
