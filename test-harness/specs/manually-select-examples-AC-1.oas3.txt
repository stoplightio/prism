====test====
Requesting an example when this example is defined, results in returning that example.
====spec====
openapi: 3.0.2
paths:
  /todos:
    get:
      responses:
        200:
          description: Get Todo Items
          "content":
            application/json:
              examples:
                cat:
                  summary: An example of a cat
                  value:
                    id: 2
                    category:
                      id: 1
                      name: Felis
                    tags:
                      - id: 1
                        name: pet
                    name: Fluffy
                    status: available
                    photoUrls: []
                dog:
                  summary: An example of a dog
                  value:
                    id: 1
                    category:
                      id: 2
                      name: Canis
                    tags:
                      - id: 2
                        name: dog
                    name: Sharik
                    status: available
                    photoUrls: []
====server====
mock -p 4010 ${document}
====command====
curl -i -X GET http://localhost:4010/todos?__example=dog
====expect====
HTTP/1.1 200 OK
content-type: application/json

{"id":1,"category":{"id":2,"name":"Canis"},"tags":[{"id":2,"name":"dog"}],"name":"Sharik","status":"available","photoUrls":[]}
