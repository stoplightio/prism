====test====
When I send a request without a body
to an operation that optionally accepts a body
I should get 200 response
====spec====
swagger: '2.0'
paths:
  /pet:
    post:
      produces:
      - application/json
      consumes:
      - application/json
      parameters:
      - in: body
        name: body
        required: false
        schema:
          type: object
          properties:
            id:
              type: integer
              format: int64
      responses:
        '200':
          schema:
            type: string
====server====
mock -p 4010 ${document}
====command====
curl -i -X POST http://127.0.0.1:4010/pet
====expect====
HTTP/1.1 200 OK
content-type: application/json
content-length: 7
Date: Fri, 21 Jun 2019 09:25:34 GMT
Connection: keep-alive

"string"
