====test====
When there are examples and an example defined, specifying example in the path results in getting back the example.
====spec====
openapi: 3.0.2
paths:
  /todos:
    get:
      responses:
        200:
          description: Get Todo Items
          "content":
            application/json:
              example:
                id: 1
                category:
                  id: 2
                  name: Canis
                tags:
                  - id: 2
                    name: dog
                name: Sharik
                status: available
                photoUrls: []
              examples:
                cat:
                  summary: An example of a cat
                  value:
                    id: 2
                    category:
                      id: 1
                      name: Felis
                    tags:
                      - id: 1
                        name: pet
                    name: Fluffy
                    status: available
                    photoUrls: []
====server====
mock -p 4010 ${document}
====command====
curl -i http://localhost:4010/todos?__example=cat
====expect====
HTTP/1.1 200 OK
content-type: application/json

{"id":2,"category":{"id":1,"name":"Felis"},"tags":[{"id":1,"name":"pet"}],"name":"Fluffy","status":"available","photoUrls":[]}
