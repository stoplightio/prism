====test====
When request JSON is invalid, Prism returns 422
====spec====
swagger: '2.0'
info:
  version: '1.0'
  title: 'abc'
paths:
  /path:
    post:
      consumes:
        - application/json
      parameters:
        - in: query
          name: param_in_query
          type: string
          required: true
        - in: body
          name: body
          schema:
            type: object
            properties:
              id:
                type: integer
              status:
                type: string
                enum:
                  - open
                  - close
            required:
              - id
      responses:
        200:
          description: 'abc'
          schema:
            type: string
====server====
mock -p 4010 ${document}
====command====
curl -i -X POST http://localhost:4010/path -H "Content-Type: application/json" --data '{ "status": "ooopsie!" }'
====expect====
HTTP/1.1 422 Unprocessable Entity
content-type: application/problem+json
Connection: keep-alive
