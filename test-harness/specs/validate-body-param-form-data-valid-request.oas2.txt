====test====
Given form data expected in request body then return OK when that body is valid.
====spec====
swagger: '2.0'
paths:
  /path:
    post:
      responses:
        200:
          schema:
            type: string
      parameters:
        - in: formData
          type: integer
          name: id
          required: true
        - in: formData
          type: string
          name: status
          required: true
          enum:
            - open
            - close
====server====
mock -p 4010
====command====
curl -i -X POST http://localhost:4010/path -H "application/x-www-form-urlencode" --data-urlencode "id=123&status=open"
====expect====
HTTP/1.1 200 OK
content-type: application/json; charset=utf-8
content-length: 0
Connection: keep-alive

"string"
